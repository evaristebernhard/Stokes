\chapter{Stokes formalization blueprint}

This blueprint records the current public proof architecture of the Lean 4
Stokes project.  It is written directly from the latest first-principles entry
point.

The current public theorem is the first-principles represented Stokes theorem
in the \texttt{CoverIndexedZeroCompactRepresentedStokesFirstPrinciplesInput}
namespace.  It proves a compact-support represented Stokes equality.  The
public input is only chartwise smoothness of the form and compactness of the
closed algebraic support.  The finite chart-box cover, pointwise half-space boxes,
support-controlled partition, refined boundary boxes, zero-localized support
proofs, local half-space Stokes fields, and represented endpoints are generated
internally.

Green nodes in the figures are implemented Lean declarations.  Blue nodes are
the remaining bridge from represented integrals to a future mathlib-native
manifold integral theorem.

\section{Current Public Theorem}
\label{sec:current_public_theorem}

\begin{definition}[First-principles input]
  \label{def:first_principles_input}
  \lean{Stokes.CoverIndexedZeroCompactRepresentedStokesFirstPrinciplesInput}
  \leanok
  The preferred public input for the represented compact-support theorem.  For
  a form \(\omega\), it exposes only:
  \[
    \texttt{chartwiseSmooth : ManifoldForm.ChartwiseSmooth I omega}
  \]
  and
  \[
    \texttt{compactSupport : IsCompact (closure (ManifoldForm.support I omega))}.
  \]
  The compact carrier is definitionally the closure of the algebraic support.
\end{definition}

\begin{definition}[Canonical represented endpoints]
  \label{def:first_principles_endpoints}
  \lean{Stokes.CoverIndexedZeroCompactRepresentedStokesFirstPrinciplesInput.canonicalRepresentedBulkIntegral,
    Stokes.CoverIndexedZeroCompactRepresentedStokesFirstPrinciplesInput.canonicalRepresentedBoundaryIntegral}
  \leanok
  \uses{def:first_principles_input}
  The bulk and boundary sides are generated from the canonical intrinsic route.
  They are represented integrals: finite sums of local coordinate integrals
  over selected half-space boxes and their true boundary faces.
\end{definition}

\begin{theorem}[First-principles represented Stokes]
  \label{thm:first_principles_represented_stokes}
  \lean{Stokes.CoverIndexedZeroCompactRepresentedStokesFirstPrinciplesInput.representedStokes}
  \leanok
  \uses{def:first_principles_input, def:first_principles_endpoints}
  The canonical represented bulk integral equals the canonical represented
  boundary integral:
  \[
    \texttt{canonicalRepresentedBulkIntegral}
      =
    \texttt{canonicalRepresentedBoundaryIntegral}.
  \]
  This is the current main theorem exposed by
  \texttt{Stokes.Global.CoverIndexedFirstPrinciples}.
\end{theorem}

\begin{definition}[Thin public import]
  \label{def:thin_public_import}
  \lean{Stokes.Global.CoverIndexedFirstPrinciples, Stokes.Global.CoverIndexed}
  \leanok
  \uses{thm:first_principles_represented_stokes}
  The public cover-indexed API imports only the first-principles theorem route.
  The default public surface is therefore the compact-support represented
  Stokes theorem rather than intermediate certificate constructors.
\end{definition}

\section{Generated Geometry}
\label{sec:generated_geometry}

\begin{definition}[Half-space model]
  \label{def:halfspace_model}
  \lean{Stokes.HalfSpaceModel}
  \leanok
  The model-with-corners range is identified with the upper half-space.  This
  gives Lean the coordinate fact needed to decide whether a self chart point is
  an interior point or a boundary point.
\end{definition}

\begin{theorem}[Self-chart pointwise boxes]
  \label{thm:pointwise_boxes_from_halfspace_model}
  \lean{Stokes.PointwiseCompactSupportChartBoxData.ofHalfSpaceModelSelfCharts}
  \leanok
  \uses{def:halfspace_model}
  For every point of the compact support carrier, self charts generate either
  an interior chart box or a boundary half-space box.  At boundary points the
  lower normal corner is zero; at interior points the box lies in the positive
  half-space.
\end{theorem}

\begin{definition}[Intrinsic input generated from first principles]
  \label{def:intrinsic_input_from_first_principles}
  \lean{Stokes.CoverIndexedZeroCompactRepresentedStokesFirstPrinciplesInput.intrinsicInput}
  \leanok
  \uses{def:first_principles_input, thm:pointwise_boxes_from_halfspace_model}
  The first-principles input constructs the intrinsic input whose fields are
  pointwise chart-box data, compactness of the carrier, chartwise smoothness,
  and support containment in the carrier.
\end{definition}

\begin{definition}[Open selected cover and partition]
  \label{def:open_selected_cover_partition}
  \lean{Stokes.CoverIndexedZeroCompactRepresentedStokesIntrinsicInput.openSelectedCover,
    Stokes.CoverIndexedZeroCompactRepresentedStokesIntrinsicInput.selectedCover,
    Stokes.CoverIndexedZeroCompactRepresentedStokesIntrinsicInput.selectedPartition}
  \leanok
  \uses{def:intrinsic_input_from_first_principles}
  Compactness of the support carrier and the pointwise chart-box neighborhoods
  generate a finite selected cover and a support-controlled partition.
\end{definition}

\begin{definition}[Selected boundary finite half-space cover]
  \label{def:selected_boundary_finite_cover}
  \lean{Stokes.CoverIndexedZeroCompactRepresentedStokesIntrinsicInput.selectedBoundaryFiniteCover}
  \leanok
  \uses{def:open_selected_cover_partition}
  Boundary active coordinate carriers are covered by finitely many selected
  half-space boxes.  These boxes are the local domains used by the boundary
  half-space Stokes theorem.
\end{definition}

\begin{definition}[Selected smooth refinement]
  \label{def:selected_smooth_refinement}
  \lean{Stokes.CoverIndexedZeroCompactRepresentedStokesIntrinsicInput.selectedSmoothRefinement,
    Stokes.CoverIndexedZeroCompactRepresentedStokesIntrinsicInput.selectedSmoothRefinement_boundaryPieces}
  \leanok
  \uses{def:selected_boundary_finite_cover}
  The selected finite cover is refined by smooth coefficients.  The refined
  boundary pieces are definitionally aligned with the selected boundary finite
  cover.
\end{definition}

\section{Zero-Localized Local Stokes}
\label{sec:zero_localized_route}

\begin{definition}[Zero-localized representative]
  \label{def:zero_localized_representative}
  \lean{Stokes.ManifoldForm.transitionPullbackInChartZero}
  \leanok
  Chart representatives are totalized by zero outside the relevant source and
  target domains.  This avoids false support claims for ordinary total chart
  representatives.
\end{definition}

\begin{theorem}[Support of selected zero-localized pieces]
  \label{thm:zero_support_selected_refinement}
  \lean{Stokes.CoverIndexedZeroCompactRepresentedStokesIntrinsicInput.zeroSupportOfSelectedSmoothRefinement}
  \leanok
  \uses{def:selected_smooth_refinement, def:zero_localized_representative}
  The zero-localized transition pullback of each selected refined localized
  form has topological support contained in the corresponding half-space support
  box.  This is the support fact that kills artificial faces in the local box
  theorem.
\end{theorem}

\begin{definition}[Interior Stokes fields from chartwise smoothness]
  \label{def:interior_fields_selected_refinement}
  \lean{Stokes.CoverIndexedZeroCompactRepresentedStokesIntrinsicInput.interiorFieldsOfSelectedSmoothRefinement}
  \leanok
  \uses{def:selected_smooth_refinement}
  Chartwise smoothness and the smooth selected refinement construct the
  analytic fields required by the half-space box theorem for each selected
  refined local piece.
\end{definition}

\begin{theorem}[Intrinsic route constructor]
  \label{thm:has_intrinsic_route_intrinsic}
  \lean{Stokes.CoverIndexedZeroCompactRepresentedStokesIntrinsicInput.hasIntrinsicRoute_intrinsic}
  \leanok
  \uses{thm:zero_support_selected_refinement,
    def:interior_fields_selected_refinement}
  The selected smooth refinement, the zero-localized support theorem, and the
  generated interior fields produce the complete intrinsic route certificate.
\end{theorem}

\begin{theorem}[Intrinsic represented Stokes]
  \label{thm:intrinsic_represented_stokes}
  \lean{Stokes.CoverIndexedZeroCompactRepresentedStokesIntrinsicInput.representedStokes}
  \leanok
  \uses{thm:has_intrinsic_route_intrinsic}
  The intrinsic input proves the represented Stokes equality before the
  first-principles wrapper hides the pointwise chart-box carrier.
\end{theorem}

\section{Euclidean And Half-Space Core}
\label{sec:euclidean_halfspace_core}

\begin{definition}[Model and manifold forms]
  \label{def:model_manifold_forms}
  \lean{Stokes.ModelForm, Stokes.ManifoldForm}
  \leanok
  A model form is a coordinate differential form, and a manifold form is a
  fiberwise alternating map on tangent spaces.
\end{definition}

\begin{theorem}[Box Stokes]
  \label{thm:box_stokes}
  \lean{Stokes.box_stokes_on_box, Stokes.box_stokes_extDeriv_smooth}
  \leanok
  \uses{def:model_manifold_forms}
  The coordinate Euclidean box theorem is repackaged for mathlib-style forms
  and \texttt{extDeriv}.
\end{theorem}

\begin{definition}[Half-space support box]
  \label{def:halfspace_support_box}
  \lean{Stokes.halfSpaceSupportBox, Stokes.boxFaceCoeffTSupportInHalfSpaceBox}
  \leanok
  Half-space support boxes encode the local geometry that separates the true
  boundary face from artificial box faces.
\end{definition}

\begin{theorem}[Half-space local Stokes]
  \label{thm:halfspace_local_stokes}
  \lean{Stokes.halfSpaceLocalStokes_compactSupport,
    Stokes.halfSpaceLocalStokes_compactSupport_of_contDiffOn_isOpen,
    Stokes.halfSpaceLocalStokes_compactSupport_of_interiorFields}
  \leanok
  \uses{thm:box_stokes, def:halfspace_support_box}
  Local Stokes on a half-space box reduces the boundary of the box to the true
  lower normal face; artificial faces vanish by support control.
\end{theorem}

\begin{definition}[Boundary sign convention]
  \label{def:boundary_sign_convention}
  \lean{Stokes.halfSpaceBoundarySign_eq_outwardFirstBoundaryOrientationSign,
    Stokes.outwardFirstBoundaryChartIntegral}
  \leanok
  \uses{thm:halfspace_local_stokes}
  The half-space boundary sign is aligned with the outward-normal-first
  convention used by boundary chart integrals.
\end{definition}

\section{Remaining Native-Integral Bridge}
\label{sec:global_assembly}

\begin{theorem}[Represented-to-native bulk integral]
  \label{thm:represented_to_native_bulk}
  \uses{thm:first_principles_represented_stokes}
  \plannedlean{Stokes.compareRepresentedBulkIntegralToMathlib}
  This future theorem should identify the canonical represented bulk integral
  with a mathlib-native integral of \(\dd\omega\), once the target manifold-form
  integral API is available or chosen.
\end{theorem}

\begin{theorem}[Represented-to-native boundary integral]
  \label{thm:represented_to_native_boundary}
  \uses{thm:first_principles_represented_stokes, def:boundary_sign_convention}
  \plannedlean{Stokes.compareRepresentedBoundaryIntegralToMathlib}
  This future theorem should identify the canonical represented boundary
  integral with the induced outward-first boundary integral of \(\omega\).
\end{theorem}

\begin{theorem}[Mathlib-native compact-support Stokes]
  \label{thm:mathlib_native_compact_support_stokes}
  \uses{thm:first_principles_represented_stokes,
    thm:represented_to_native_bulk, thm:represented_to_native_boundary}
  \plannedlean{Stokes.globalStokes_compactSupport}
  After both comparison theorems are available, the represented theorem becomes
  the native compact-support Stokes statement
  \[
    \int_M \dd\omega = \int_{\partial M} \omega.
  \]
\end{theorem}

\begin{theorem}[Compact manifold Stokes]
  \label{thm:compact_manifold_stokes}
  \uses{thm:mathlib_native_compact_support_stokes}
  \plannedlean{Stokes.globalStokes_compactManifold}
  The compact-support theorem specializes to compact oriented smooth manifolds
  with boundary by taking the support carrier to be the whole compact manifold.
\end{theorem}
